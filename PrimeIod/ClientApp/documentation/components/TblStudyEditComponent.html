import { Component, OnInit, Input, EventEmitter, Output } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { Location } from '@angular/common';
import { FormGroup, FormBuilder, Validators, FormControl } from '@angular/forms';
// import { NgbActiveModal } from '@ng-bootstrap/ng-bootstrap';
import { TblStudyService } from '../../tbl-study.service';
import { StudyEditService } from './study-edit.service';
// import { stringify } from 'querystring';

@Component({
  selector: 'app-tbl-study-edit',
  templateUrl: './tbl-study-edit.component.html',
  styleUrls: ['./tbl-study-edit.component.scss'],
  providers: [StudyEditService]
})
export class TblStudyEditComponent implements OnInit {
  // @Input() tblStudyId: string;
  // @Input() action = '';
  tblStudy: any;
  // tblStudyForm: FormGroup;

  tblStudyId: 0;

  constructor(
    private actRoute: ActivatedRoute,
    private formBuilder: FormBuilder,
    // public activeModal: NgbActiveModal,
    private tblStudyService: TblStudyService,
    private studyEditService: StudyEditService,
    private location: Location,
    private router: Router
  ) {
    this.tblStudyId = this.actRoute.snapshot.params.id;
  }
  study: any;

  paramList: any;
  ngOnInit() {
    this.studyEditService.loadParams();
    this.loadParamList();
    this.studyEditService.loadStudy(this.tblStudyId);

    //Example navigate
    let xyz = this.studyEditService._study.subscribe((study: any) => {
      this.study = study;
      //for testing:
      // this.testNavigate();  //works if called inside the subscribe.
    });
    this.load();
  }

  loadParamList() {
    let xyz = this.studyEditService.paramList.subscribe((paramList: any) => (this.paramList = paramList));
  }

  load() {
    let xyz = this.studyEditService._study.subscribe((study: any) => {
      this.study = study;
      //for testing:
      this.testNavigate();
    });
  }

  save() {
    this.studyEditService.saveStudy();
    this.router.navigate(['/study-group/tbl-study'], {
      queryParams: { refresh: new Date().getTime(), relativeTo: this.actRoute.parent }
    });

    //    this.router.navigate(['/study-group/tbl-edit', {id: studyId }],{ relativeTo: this.actRoute.parent});

    // this.location.back();

    // if (this.action == 'Add') {

    // if (this.action == 'Edit') {
    //   // this.updateTblStudy();
    // }
  }

  back() {
    this.router.navigate(['/study-group/tbl-study'], { queryParams: { relativeTo: this.actRoute.parent } });
  }

  testNavigate() {
    //http://localhost:4200/study-group/tbl-edit/393/study-assumptions
    this.router.navigate(
      ['study-assumptions'],
      { relativeTo: this.actRoute }
      // {queryParams: {relativeTo: this.actRoute.parent}}
    );
  }
}
