import { Component, OnInit } from '@angular/core';
import { FormGroup, FormBuilder, Validators, FormControl } from '@angular/forms';
import { StudyEditService } from '../study-edit.service';

@Component({
  selector: 'app-assumptions',
  templateUrl: './assumptions.component.html',
  styleUrls: ['./assumptions.component.css']
})
export class AssumptionsComponent implements OnInit {
  study: any;
  studyAssumptions: any;

  paramList: any;
  cdmsParams: any;
  assumptionsForm: FormGroup;
  constructor(private studyEditService: StudyEditService, private formBuilder: FormBuilder) {}

  ngOnInit(): void {
    this.buildForm();
    this.studyEditService.paramList.subscribe((params: any) => (this.paramList = params));
    this.studyEditService._study.subscribe((st: any) => {
      this.study = st;
      this.studyAssumptions = st?.studyAssumptions;

      if (this.studyAssumptions != null) {
        this.assumptionsForm.patchValue(this.studyAssumptions);
      }

      // this.assumptionsForm.controls.recId.patchValue(this.studyAssumptions?.recId);
      // this.assumptionsForm.controls.totalUniqueForms.patchValue(this.studyAssumptions?.totalUniqueForms);
      // this.assumptionsForm.controls.protocolPhasePid.patchValue(this.studyAssumptions?.protocolPhasePid);
      // this.assumptionsForm.controls.protocolComplexityPid.patchValue(this.studyAssumptions?.protocolComplexityPid);
    });

    this.assumptionsForm.valueChanges.subscribe(studyAssumptions => (this.study.studyAssumptions = studyAssumptions));
  }

  buildForm() {
    this.assumptionsForm = this.formBuilder.group({
      recId: new FormControl(''),
      totalUniqueForms: new FormControl(''),
      protocolPhasePid: new FormControl(''),
      protocolComplexityPid: new FormControl('')
    });
  }

  getDropdownValues(parId: number) {
    let values = Array.from(this.paramList);
    return values.filter((e: any) => e.parId === parId);
  }
}
