import { Injectable } from '@angular/core';
// import {Observable} from "rxjs/Observable";
// import {Subject} from "rxjs/Subject";

import { BehaviorSubject } from 'rxjs/internal/BehaviorSubject';
import { TblStudyService } from '../../tbl-study.service';
import { TblParamService } from '@app/prism/masters/tbl-param/tbl-param.service';
import { Study } from '../../../study';

// @Injectable({
//   providedIn: 'root'
// })
@Injectable()
export class StudyEditService {
  private studySource: BehaviorSubject<Study> = new BehaviorSubject(this.initialState);
  _study = this.studySource.asObservable();

  private paramListSource: BehaviorSubject<any> = new BehaviorSubject(this.initialParamState);
  paramList = this.paramListSource.asObservable();

  constructor(private tblStudyService: TblStudyService, private tblParamService: TblParamService) {}

  get initialState() {
    return <any>{};
  }

  get initialParamState() {
    return <any>{};
  }

  get study() {
    return this.studySource.next(this.initialState);
  }

  loadStudy(id: number) {
    this.tblStudyService.getTblStudy(id).subscribe((study: any) => {
      this.studySource.next(study);
    });
  }

  loadParams() {
    this.tblParamService.getParamList().subscribe((params: any) => {
      this.paramListSource.next(params);
    });
  }

  // updateStudy(study: any){
  //   this.modifiedStudySource.next(study);
  // }

  // get modifiedStudy(){
  //   return this._modifiedStudy;
  // }
  saveStudy() {
    let st: any;
    this._study.subscribe(abc => (st = abc));
    console.log('saveStudy' + JSON.stringify(st));
    this.tblStudyService.updateTblStudy(st).subscribe(
      res => {
        console.log(`success`);
      },
      err => {
        console.log(`error while editing = ${err}`);
      }
    );
  }
}
