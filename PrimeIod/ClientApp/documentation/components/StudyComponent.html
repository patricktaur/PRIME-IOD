import { Component, OnInit, forwardRef } from '@angular/core';

import {
  ControlValueAccessor,
  FormControl,
  NG_VALUE_ACCESSOR,
  NG_VALIDATORS,
  FormGroup,
  Validator,
  Validators,
  AbstractControl,
  ValidationErrors,
  FormBuilder
} from '@angular/forms';
import { StudyEditService } from '../study-edit.service';
// import {Study} from '../study'

@Component({
  selector: 'app-study',
  templateUrl: './study.component.html',
  styleUrls: ['./study.component.css'],

  providers: [
    {
      provide: NG_VALUE_ACCESSOR,
      useExisting: forwardRef(() => StudyComponent),
      multi: true
    },
    {
      provide: NG_VALIDATORS,
      useExisting: forwardRef(() => StudyComponent),
      multi: true
    }
  ]
})
export class StudyComponent implements OnInit, ControlValueAccessor, Validator {
  study: any;
  constructor(private studyEditService: StudyEditService) {}

  ngOnInit() {
    this.studyEditService._study.subscribe((st: any) => (this.study = st));
  }

  public studyForm: FormGroup = new FormGroup({
    firstIconNumber: new FormControl('', [Validators.required]),
    secondIconNumber: new FormControl('', [Validators.required])
  });

  public onTouched: () => void = () => {};

  writeValue(val: any): void {
    val && this.studyForm.setValue(val, { emitEvent: false });
  }
  registerOnChange(fn: any): void {
    this.studyForm.valueChanges.subscribe(fn);
  }
  registerOnTouched(fn: any): void {
    this.onTouched = fn;
  }
  setDisabledState?(isDisabled: boolean): void {
    isDisabled ? this.studyForm.disable() : this.studyForm.enable();
  }

  validate(c: AbstractControl): ValidationErrors | null {
    return this.studyForm.valid ? null : { invalidForm: { valid: false, message: 'basicInfoForm fields are invalid' } };
  }
}
