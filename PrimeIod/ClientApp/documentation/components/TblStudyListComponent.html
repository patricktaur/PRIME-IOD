  loadReport() {
    this.isLoading = true;
    this.tblStudyService.getTblStudyList().subscribe(
      (res: any) => {
        if (res.status === 400) {
          return;
        } else {
          this.records = res;
          if (this.records) {
            this.filteredStudies = this.records;
          }
        }
        this.isLoading = false;
      },
      (err: any) => {
        console.log(`err = ${JSON.stringify(err, null, 2)}`);
        this.isLoading = false;
      }
    );
  }

  onSearchTextChange(searchText: string) {
    this.searchTerm = searchText;
  }

  onPageChange(pageOptions: any) {
    this.pageSize = pageOptions.pageSize;
    this.pageNumber = pageOptions.page;
  }

  onFilterChange(filteredRecords: any) {
    this.filteredStudies = filteredRecords;
  }

  // For 'Add' option
  addStudy() {
    this.router.navigate(['/study-group/tbl-edit', { id: 0 }], { relativeTo: this.actRoute.parent });
  }

  editStudy(studyId: number) {
    this.router.navigate(['/study-group/tbl-edit', studyId], { relativeTo: this.actRoute.firstChild });
  }

  //for delete option
  deleteTblStudy(tblStudy: any) {
    this.tblStudyService.deleteTblStudy(tblStudy).subscribe(
      res => {
        this.toastService.show('TblStudy deleted successfully.', { classname: 'bg-success text-light', delay: 3000 });
        this.loadReport();
      },
      err => {
        console.log(`Error while deleting. = ${err}`);
      }
    );
  }

  confirmDeleteTblStudy(tblStudy: any) {
    const modalRef = this.modalService.open(ConfirmationModalComponent, { ariaLabelledBy: 'modal-basic-title' });
    modalRef.componentInstance.confirmationText = `Are you sure you want to delete ${tblStudy.firstIconNumber} ?`; //todo: replace firstIconNumber with field name

    modalRef.result.then(
      result => {
        console.log(`result = ${result}`);
        if (result === 'confirm') {
          this.deleteTblStudy(tblStudy);
        }
      },
      err => {
        console.log(`err = ${JSON.stringify(err)}`);
      }
    );
  }
